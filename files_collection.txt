    <script>
        let phaseCount = 1;
        let chart;

        function addPhase() {
            if (phaseCount >= 5) {
                alert("Maximum of 5 phases allowed.");
                return;
            }
            phaseCount++;
            const newPhase = document.createElement('div');
            newPhase.className = 'phase';
            newPhase.id = `phase${phaseCount}`;
            newPhase.innerHTML = `
                <h3>Phase ${phaseCount}</h3>
                <label for="startingAmount${phaseCount}">Starting Amount ($):</label>
                <input type="number" id="startingAmount${phaseCount}" min="0" step="0.01" required disabled><br>
                <label for="years${phaseCount}">Years to Invest:</label>
                <input type="number" id="years${phaseCount}" min="0" step="1" required><br>
                <label for="monthlyContribution${phaseCount}">Monthly Contribution ($):</label>
                <input type="number" id="monthlyContribution${phaseCount}" min="0" step="0.01" required><br>
                <label for="annualReturn${phaseCount}">Annual Rate of Return (%):</label>
                <input type="number" id="annualReturn${phaseCount}" min="0" step="0.01" required><br>
                <button class="delete-btn" onclick="deletePhase(${phaseCount})">Delete Phase</button>
            `;
            document.getElementById('phases').appendChild(newPhase);
        }

        function deletePhase(phaseNumber) {
            if (phaseNumber === 1) {
                alert("Cannot delete the first phase.");
                return;
            }
            document.getElementById(`phase${phaseNumber}`).remove();
            phaseCount--;
            // Renumber remaining phases
            for (let i = phaseNumber + 1; i <= phaseCount + 1; i++) {
                let phase = document.getElementById(`phase${i}`);
                if (phase) {
                    phase.id = `phase${i-1}`;
                    phase.querySelector('h3').textContent = `Phase ${i-1}`;
                    phase.querySelector('.delete-btn').setAttribute('onclick', `deletePhase(${i-1})`);
                }
            }
        }

        function calculate() {
            let totalAmount = 0;
            let totalPrincipal = 0;
            let steps = [];
            let chartData = [];
            let cumulativeYears = 0;

            for (let i = 1; i <= phaseCount; i++) {
                const startingAmountInput = document.getElementById(`startingAmount${i}`);
                const startingAmount = i === 1 ? parseFloat(startingAmountInput.value) || 0 : totalAmount;
                startingAmountInput.value = startingAmount.toFixed(2);
                
                const years = parseInt(document.getElementById(`years${i}`).value) || 0;
                const monthlyContribution = parseFloat(document.getElementById(`monthlyContribution${i}`).value) || 0;
                const annualReturn = parseFloat(document.getElementById(`annualReturn${i}`).value) || 0;

                const monthlyRate = annualReturn / 100 / 12;

                let phaseStartPrincipal = totalPrincipal;
                totalAmount = startingAmount;

                steps.push(`Phase ${i}:`);
                steps.push(`  Starting Amount: $${startingAmount.toFixed(2)}`);
                steps.push(`  Years: ${years}`);
                steps.push(`  Monthly Contribution: $${monthlyContribution.toFixed(2)}`);
                steps.push(`  Annual Return: ${annualReturn}%`);

                for (let year = 1; year <= years; year++) {
                    let yearlyPrincipal = phaseStartPrincipal + monthlyContribution * year * 12;
                    let yearlyAmount = totalAmount;

                    for (let month = 1; month <= 12; month++) {
                        yearlyAmount = (yearlyAmount + monthlyContribution) * (1 + monthlyRate);
                    }

                    chartData.push({
                        x: cumulativeYears + year,
                        y: yearlyAmount.toFixed(2),
                        principal: yearlyPrincipal.toFixed(2),
                        gains: (yearlyAmount - yearlyPrincipal).toFixed(2)
                    });

                    totalAmount = yearlyAmount;
                }

                totalPrincipal += monthlyContribution * years * 12;

                steps.push(`  Ending Amount: $${totalAmount.toFixed(2)}`);
                steps.push(`  Total Principal at End: $${totalPrincipal.toFixed(2)}`);
                steps.push(`  Total Gain: $${(totalAmount - totalPrincipal).toFixed(2)}`);
                steps.push('');

                cumulativeYears += years;
            }

            document.getElementById('result').innerHTML = `<strong>Total after ${phaseCount} phase(s): $${totalAmount.toFixed(2)}</strong><br>` +
                                                         `<strong>Total Principal: $${totalPrincipal.toFixed(2)}</strong><br>` +
                                                         `<strong>Total Gain: $${(totalAmount - totalPrincipal).toFixed(2)}</strong>`;
            document.getElementById('steps').innerHTML = '<h3>Calculation Steps:</h3>' + steps.join('<br>');

            updateChart(chartData);
        }

        function updateChart(data) {
            const ctx = document.getElementById('investmentChart').getContext('2d');
            
            if (chart) {
                chart.destroy();
            }

            chart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: data.map(d => `Year ${d.x}`),
                    datasets: [
                        {
                            label: 'Principal',
                            data: data.map(d => d.principal),
                            backgroundColor: 'rgba(75, 192, 192, 0.6)',
                            stack: 'Stack 0',
                        },
                        {
                            label: 'Investment Gains',
                            data: data.map(d => d.gains),
                            backgroundColor: 'rgba(153, 102, 255, 0.6)',
                            stack: 'Stack 0',
                        },
                        {
                            label: 'Total Value',
                            data: data.map(d => d.y),
                            type: 'line',
                            borderColor: 'rgba(255, 99, 132, 1)',
                            borderWidth: 2,
                            fill: false,
                            tension: 0.1
                        }
                    ]
                },
                options: {
                    responsive: true,
                    scales: {
                        x: {
                            stacked: true,
                            title: {
                                display: true,
                                text: 'Year'
                            }
                        },
                        y: {
                            stacked: true,
                            title: {
                                display: true,
                                text: 'Investment Value ($)'
                            }
                        }
                    }
                }
            });
        }

            <h3>Watts</h3>
            <p>Watts measure power output - the rate at which energy is being used. In rowing, it's a direct measure of how hard you're working. Higher watts mean more power per stroke and generally faster splits.</p>
        </div>

        <div class="metric-explanation">
            <h3>Stroke Rate</h3>
            <p>Measured in strokes per minute (spm), stroke rate indicates how many complete stoke cycles you do per minute.</p>
        </div>

        <div class="metric-explanation">
            <h3>Joules</h3>
            <p>Joules measure the energy per stroke. It's calculated as (Watts × 60) ÷ Stroke Rate. This tells you how much energy you're putting into each individual stroke, regardless of how fast you're taking the strokes. Higher joules typically indicate stronger, more efficient strokes.</p>
        </div>

    <script>
        function calculate() {
            const strokeRate = parseFloat(document.getElementById('strokeRate').value);
            const watts = parseFloat(document.getElementById('watts').value);

            if (!strokeRate || !watts) return;

            // Calculate time per stroke (60/strokeRate)
            const timePerStroke = 60 / strokeRate;

            // Calculate joules ((watts * 60) / strokeRate)
            const joules = (watts * 60) / strokeRate;

            // Display results
            document.getElementById('timePerStroke').textContent = timePerStroke.toFixed(2) + ' seconds';
            document.getElementById('joules').textContent = joules.toFixed(2);
            document.getElementById('time2k').textContent = '6:07.0';  // Need formula
            document.getElementById('time5k').textContent = '15:17.5'; // Need formula
        }
    </script>

        function generateSplitInputs() {
            const distance = parseInt(document.getElementById('splitDistance').value);
            const numSplits = Math.ceil(2000 / distance);
            const container = document.getElementById('splitInputs');
            
            container.innerHTML = '';
            for (let i = 0; i < numSplits; i++) {
                const startDist = i * distance;
                const endDist = Math.min(startDist + distance, 2000);
                const isLastSplit = endDist === 2000;
                
                const div = document.createElement('div');
                div.className = 'split-row';
                const splitHtml = `
                    <label>${startDist}m to ${endDist}m:</label>
                    <input type="number" step="0.1" class="split-input" value="${getSavedValue(i)}">
                    ${isLastSplit ? '<button onclick="toggleSprintMode()">Plan Sprint</button>' : ''}
                `;
                div.innerHTML = splitHtml;
                container.appendChild(div);
            }
            inSprintMode = false;
        }

        function getSavedValue(index) {
            const inputs = document.getElementsByClassName('split-input');
            return inputs[index]?.value || '';
        }

        function toggleSprintMode() {
            const distance = parseInt(document.getElementById('splitDistance').value);
            const sprintStart = 2000 - distance; // 1500 for 500m, 1600 for 400m
            const lastSplitRow = document.querySelector('.split-row:last-child');
            
            if (!inSprintMode) {
                // Enter sprint mode
                const sprintSection = document.createElement('div');
                sprintSection.className = 'sprint-section';
                sprintSection.innerHTML = `
                    <h4>Sprint Breakdown (${sprintStart}m to 2000m)</h4>
                    ${generateSprintInputs(sprintStart)}
                    <button onclick="applySprintSplits()">Apply Sprint</button>
                    <button onclick="toggleSprintMode()" class="back-btn">Back</button>
                `;
                lastSplitRow.classList.add('sprint-active');
                lastSplitRow.appendChild(sprintSection);
                inSprintMode = true;
            } else {
                // Exit sprint mode
                const sprintSection = document.querySelector('.sprint-section');
                if (sprintSection) sprintSection.remove();
                lastSplitRow.classList.remove('sprint-active');
                inSprintMode = false;
            }
        }

        function generateSprintInputs(startDist) {
            let html = '';
            for (let i = startDist; i < 2000; i += 100) {
                html += `
                    <div class="input-group">
                        <label>${i}m to ${i + 100}m:</label>
                        <input type="number" step="0.1" class="sprint-split">
                    </div>
                `;
            }
            return html;
        }

        function applySprintSplits() {
            const sprintSplits = Array.from(document.getElementsByClassName('sprint-split'))
                                    .map(input => parseFloat(input.value))
                                    .filter(val => !isNaN(val));
            
            if (sprintSplits.length === 0) return;
            
            // Calculate weighted average for final split
            const avgSprint = sprintSplits.reduce((a, b) => a + b, 0) / sprintSplits.length;
            
            // Apply to last main split
            const mainSplits = document.getElementsByClassName('split-input');
            mainSplits[mainSplits.length - 1].value = avgSprint.toFixed(1);
            
            calculateAverages();
        }

        function calculateAverages() {
            const splits = Array.from(document.getElementsByClassName('split-input'))
                               .map(input => parseFloat(input.value) || 0);
            const distance = parseInt(document.getElementById('splitDistance').value);
            
            if (splits.some(s => s <= 0)) return;

            let runningAvgHtml = '';
            let totalTime = 0;
            let totalDistance = 0;

            splits.forEach((split, i) => {
                const currentDistance = Math.min(distance, 2000 - totalDistance);
                totalDistance += currentDistance;
                totalTime += split;
                
                const avgSplit = totalTime / (i + 1);
                const displayMin = Math.floor((avgSplit + 60) / 60);
                const displaySec = ((avgSplit + 60) % 60).toFixed(1);
                
                runningAvgHtml += `
                    <p>Average to ${totalDistance}m: 
                       <strong>${displayMin}:${displaySec.padStart(4, '0')}</strong>
                    </p>`;
            });

            document.getElementById('runningAverages').innerHTML = runningAvgHtml;

            const finalAvgSplit = totalTime / splits.length;
            const finalTotalSeconds = (finalAvgSplit + 60) * 4;
            const finalMinutes = Math.floor(finalTotalSeconds / 60);
            const finalSeconds = (finalTotalSeconds % 60).toFixed(1);
            
            document.getElementById('finalTime').innerHTML = 
                `<strong>${finalMinutes}:${finalSeconds.padStart(4, '0')}</strong>`;
        }

        // Initialize
        generateSplitInputs();
    </script>

            <p>This <a href="https://colab.research.google.com/drive/1lLvYMplimAMliTnviatqF0Iv8CesA-Qf?usp=sharing"> implementation </a> builds a language model from scratch, with focus on replicating autoregressive modeling similar to GPT-2. The goal is to break down the abstraction layers typically found in modern frameworks and examine each step in detail. Each code block is designed to be self-contained, and it provides explanations and intuitions behind key matrix operations.</p>

            <p><b>Attention is a key mechanism in transformer models</b>, designed to allow words to update their meanings based on context. Initially, words are represented as high-dimensional vectors called <em>embeddings</em>. These embeddings only encode the meaning of individual words, but the goal of attention is to refine them to incorporate contextual information.</p>

            <p>The attention mechanism revolves around three main components: <em>query</em>, <em>key</em>, and <em>value</em> matrices. These are learnable parameters that transform the initial embeddings. Conceptually (though not always empirically), the query matrix produces "questions" for each word, while the key matrix generates potential "answers." The interaction between these queries and keys forms the basis of determining which words are relevant to updating the meaning of others.</p> 

            <p>Consider the sentence "The ancient oak tree provided cool shade on the hot summer day." The query matrix might transform the embedding of "tree" into a vector that asks, "What are my defining characteristics?" Meanwhile, the key matrix might transform the embeddings of other words into vectors that potentially answer this query. To determine the relevance of each word to "tree," the model computes the dot product between the query vector for "tree" and the key vectors for all other words. This calculation results in a grid of attention scores.</p>

            <p>For instance, the score between "ancient" and "tree" would likely be high, as would the score between "oak" and "tree," indicating strong relationships. Conversely, the scores between "tree" and words like "hot" or "summer" would likely be lower, as they're less directly related to the tree's characteristics. These scores are then normalized using a softmax function, converting them into weights between 0 and 1, creating the attention pattern. In this pattern, "ancient" and "oak" would have high weights with respect to "tree," allowing the model to update the embedding of "tree" to reflect that it's specifically an old oak tree. Meanwhile, words like "hot" and "summer" would have lower weights, contributing less to the updated meaning of "tree."</p>

            <p>The value matrix comes into play for actually updating the embeddings.  Unlike the query and key matrices, which are used to compute relevance, the value matrix transforms the original embeddings into "value vectors" that contain the actual information used for updates. This transformation serves several purposes: it allows the model to learn what information is most useful for updates, provides more flexibility in representing information, and can help manage computational complexity by potentially reducing dimensionality. Instead of directly using the original embeddings, which contain a lot of information that may not all be relevant for updates, the value vectors focus on the most important aspects for updating word meanings in context.</p>
            
            <p>To update a given word's embedding (like "tree"), the model uses the weights from the attention pattern to compute a weighted sum of all value vectors. This process works as follows: First, we have value vectors for all words in the sentence. We also have attention weights showing how relevant each word is to "tree". The model then multiplies each value vector by its corresponding attention weight and sums up all these weighted value vectors. For instance, the value vectors of "ancient" and "oak" would contribute more heavily to updating "tree" due to their higher attention weights, while "hot" and "summer" would contribute less. This weighted sum represents the contextual information to be added to the original embedding of "tree", resulting in a new, context-aware embedding that reflects its specific characteristics in this sentence. This process is repeated for every word, allowing each to be updated based on its relevant context as determined by the attention pattern, using these specially computed value vectors instead of directly using other words' embeddings.</p>

            <p>Another important aspect of attention in language models is masking. During training, the model needs to predict the next word based only on previous words. To prevent future words from influencing earlier ones, the attention mechanism applies a mask that effectively sets the relevance of future words to zero.</p>

            <p>What I explained so far happened in a single attention head. Instead of performing a single attention operation, transformer models typically run multiple attention heads in parallel. Each head has its own set of query, key, and value matrices, allowing the model to capture different types of relationships between words simultaneously. The attention mechanism is applied iteratively in transformer models. Embeddings go through multiple attention blocks, each time becoming more refined and context-aware. This allows the model to capture increasingly complex and abstract relationships within the text.</p>

            <p>I keep thinking about how strange it is that something could know everything about the letter "H": its shape, its sound, how it fits into "hello," but never actually experience it. What does it even mean to learn something without sensing it?</p>

            <p>I think about <a href="https://en.wikipedia.org/wiki/Knowledge_argument" class="back-link">Mary's Room</a> a lot. This scientist who's lived her whole life in a black and white room. She knows everything about color, wavelengths, light reflections, how our brains process different colors, but she's never seen color herself. When she finally steps out and sees red for the first time, something changes. There's a kind of knowledge that only comes through direct experience, something that all her intellectual understanding could never give her.</p>

            <p>Language models like ChatGPT remind me of Mary before she sees color. They process letters, words, and patterns at an incredible level, but their understanding is purely diegetic, stuck inside the internal, propositional content of words. They know about the letter "H," how it fits in the alphabet, how it works in sentences, the statistics of how it appears. But can a model that has never seen the letter "H" in the real world, never interacted with its shape or seen it in context, ever really understand it like we do?</p>

            <p>This leads us to the idea of supradiegetic knowledge—understanding that goes beyond just the internal logic of language and connects with how it actually exists in the world. It's the difference between knowing "H" is the eighth letter of the alphabet and knowing what it's like to see "H" on a page, spot it in your peripheral vision, tell it apart from other letters at a glance. That physical relationship with language is something we take for granted, but it's exactly what models like ChatGPT don't have.</p>

            <p>The gap between diegetic and supradiegetic knowledge feels fundamental. Just like Mary can describe color but not <em>know</em> it until she sees it, language models can process words without ever really getting the sensory aspects that make those words meaningful. This raises deeper questions about what it means to understand language at all. For us, language is something we experience through sight, sound, and touch. For a model, it's just patterns of tokens, like seeing only the shadow of what we actually experience every day.</p>

            <p>Looking at it this way, LLMs like ChatGPT are powerful but incomplete. They exist in this kind of linguistic limbo where they can work with words and sentences but miss the full picture of what language is. And like Mary, they stay in the dark unless we can somehow give them <em>true</em> sensory experience, if that's even possible at all.</p>

// components/gallery.js

document.addEventListener('DOMContentLoaded', function() {
    // Create and append gallery HTML
    const galleryHTML = `
        <div class="photo-gallery">
            <div class="gallery-header">
                <span class="gallery-title">Featured moments</span>
                <div class="thumbnail-container">
                    <div class="thumbnail" data-full="../pictures/gallery/dec-2021-pohang.jpg" data-caption="Marine Corps, Pohang, December 2021">
                        <img src="../pictures/dec-2021-pohang-thumb.jpg" alt="Marine Corps">
                    </div>
                    <div class="thumbnail" data-full="pictures/princeton-chase-2024.jpg" data-caption="Penn Lightweight Rowing at Princeton Chase">
                        <img src="pictures/princeton-chase-2024-thumb.jpg" alt="Penn Rowing">
                    </div>
                </div>
            </div>
        </div>

        <div id="photo-modal" class="modal">
            <div class="modal-content">
                <span class="close-button">&times;</span>
                <img id="modal-image" src="" alt="">
                <p id="modal-caption" class="modal-caption"></p>
            </div>
        </div>
    `;

    // Insert gallery before the "Get in Touch" section
    const getInTouchSection = document.querySelector('h3');
    getInTouchSection.parentNode.insertBefore(
        createElementFromHTML(galleryHTML),
        getInTouchSection
    );

    // Add event listeners
    const thumbnails = document.querySelectorAll('.thumbnail');
    const modal = document.getElementById('photo-modal');
    const modalImg = document.getElementById('modal-image');
    const modalCaption = document.getElementById('modal-caption');
    const closeButton = document.querySelector('.close-button');

    thumbnails.forEach(thumb => {
        thumb.addEventListener('click', function() {
            modal.style.display = "flex";
            modalImg.src = this.dataset.full;
            modalCaption.textContent = this.dataset.caption;
        });
    });

    closeButton.addEventListener('click', () => {
        modal.style.display = "none";
    });

    modal.addEventListener('click', (e) => {
        if (e.target === modal) {
            modal.style.display = "none";
        }
    });

    // Helper function to create element from HTML string
    function createElementFromHTML(htmlString) {
        const div = document.createElement('div');
        div.innerHTML = htmlString.trim();
        return div.firstChild;
    }
});
